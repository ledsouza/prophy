## Usage

```jsx
import ComboBox from "./ComboBox";

const MyComponent = () => {
    const data = [
        /* your data here */
    ];
    const [selectedValue, setSelectedValue] = useState(null);

    return (
        <ComboBox
            data={data}
            selectedValue={selectedValue}
            onChange={setSelectedValue}
            placeholder="Select an option"
        >
            Label
        </ComboBox>
    );
};
```
